function PreloadFiles takes nothing returns nothing

	call PreloadStart()
	call Preload( "<!-- Chat Commands generated with chat command "-website". -->" )
	call Preload( "<ul>" )
	call Preload( "<li>"-help/-h": Lists chat commands.</li>" )
	call Preload( "<li>"-i/-info/-stats X": Shows info about player X (number, color, name).</li>" )
	call Preload( "<li>"-r/-revive": Selects your altar to revive your hero.</li>" )
	call Preload( "<li>"-discord": Shows the Discord server URL.</li>" )
	call Preload( "<li>"-h1/-h2/-h3": Allows you to (re)pick your heroes.</li>" )
	call Preload( "<li>"-p1/-p2": Allows you to (re)pick your professions.</li>" )
	call Preload( "<li>"-r1/-r2": Allows you to (re)pick your races.</li>" )
	call Preload( "<li>"-repick": Allows you to choose a new hero but removes all of your units from the map!</li>" )
	call Preload( "<li>"-fullrepick": Lose 3/4 of your XP and Evolution but repick the game mode.</li>" )
	call Preload( "<li>"-unlock/-lock": Allows you to unlock everything in the map.</li>" )
	call Preload( "<li>"-nobossicons/-bossicons": Disables/enables Boss icons on the minimap.</li>" )
	call Preload( "<li>"-auto": Auto skills all heroes in selection.</li>" )
	call Preload( "<li>"-enchanter": Lists all items which can be enchanted.</li>" )
	call Preload( "</ul>" )
	call Preload( "<!-- Chat Commands generated with chat command "-website". -->" )
	call PreloadEnd( 0.0 )

endfunction
