function PreloadFiles takes nothing returns nothing

	call PreloadStart()
	call Preload( "<!-- Chat Commands generated with chat command "-website". -->" )
	call Preload( "<ul>" )
	call Preload( "<li>"-help/-h": Lists chat commands.</li>" )
	call Preload( "<li>"-i/-info/-stats X": Shows info about player X (number, color, name).</li>" )
	call Preload( "<li>"-r/-revive": Selects your altar to revive your hero.</li>" )
	call Preload( "<li>"-discord": Shows the Discord server URL.</li>" )
	call Preload( "<li>"-h1/-h2/-h3": Allows you to (re)pick your heroes.</li>" )
	call Preload( "<li>"-p1/-p2": Allows you to (re)pick your professions.</li>" )
	call Preload( "<li>"-r1/-r2": Allows you to (re)pick your races.</li>" )
	call Preload( "</ul>" )
	call Preload( "<!-- Chat Commands generated with chat command "-website". -->" )
	call PreloadEnd( 0.0 )

endfunction
